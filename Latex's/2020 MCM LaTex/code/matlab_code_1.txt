X1=[ones(size(dryer_review)),dryer_review];
y=dryer_rating;
[b,bint,r,rint,stats]=regress(y,X1);
z1=b(1)+b(2)*dryer_rating;
figure;
plot(dryer_review,y,'r+',dryer_rating,z1,'b');
xlabel('reformed review value'),ylabel('star rating');
title('hair dryer')
grid on;
